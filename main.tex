\begin{question}{2}
    Tìm $x$ của phương trình : \[x + 1 = 2\]
\end{question}

\begin{solution}
    \begin{alignat*}{2}
        x + 1 &= 2 \\
        \Leftrightarrow x &= 2 - 1 \\
        \Leftrightarrow x &= 1
    \end{alignat*}
    Vậy nghiệm của phương trình là $x = 1$\qed  
\end{solution}
