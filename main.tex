\begin{question}{1}
\[1 + 1 = ?\]
\end{question}

\begin{proof}
Dễ dàng ta thấy : \highlight{$1 + 1 = 2$}
\end{proof}

\begin{question}{2}
Tìm $x$ của phương trình : \[x + 1 = 2\]
\end{question}

\begin{proof}
\begin{alignat*}{2}
    && x + 1 &= 2 \\
    &\Leftrightarrow& x &= 2 - 1 \\
    &\Leftrightarrow& x &= 1
\end{alignat*}
\end{proof}
